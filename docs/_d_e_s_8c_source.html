157 * @brief Generate 16 subKeys with the original DES key.
158 * @param originalKey The original DES key.
159 * @return DES_Key* A pointer to an array of 17 keys: 1 original key and 16 subKeys.

161 DES_Key* DES_generateSubKeys(const DES_Key originalKey)
162 {
163  static DES_Key subKeys[17];
164 
165  uint8_t index = 0, pos = 0, shiftLen;
166  uint32_t bit = 0;
167 
168  // Generate initial permuted key K_0
169  subKeys[0].key = 0x0;
170  for (; index < 56; index++) {
171  pos = __permuted_key_table_PC1__[index];
172  if (pos < 32) { // get bit from low 32bit of originalKey
173  bit = __getBitAt__(__low32__(originalKey), pos);
174  } else { // get bit from hight 32bit of originalKey
175  bit = __getBitAt__(__high32__(originalKey), pos - 32);
176  }
177  if (index < 24) {
178  __high32__(subKeys[0]) <<= 1;
179  __high32__(subKeys[0]) |= (bit & 0x1);
180  }
181  else {
182  __low32__(subKeys[0]) <<= 1;
183  __low32__(subKeys[0]) |= (bit & 0x1);
184  }
185  }
186  // Generate C_0 and D_0
187  DES_Key_pair keyPairGroup[17];
188  keyPairGroup[0].dKey = (__low32__(subKeys[0]) & 0xFFFFFFF);
189  keyPairGroup[0].cKey = __high32__(subKeys[0]);
190  keyPairGroup[0].cKey <<= 4;
191  keyPairGroup[0].cKey |= (__low32__(subKeys[0]) >> 28);
192  // Generate cKey and dKey for K_1 ~ K_16
193  for (index = 1; index < 17; index++) {
194  shiftLen = __keyPair_leftshift_table__[index];
195  keyPairGroup[index].cKey = (keyPairGroup[index - 1].cKey << (4 + shiftLen)) >> 4;
196  keyPairGroup[index].cKey |= (keyPairGroup[index - 1].cKey >> (28 - shiftLen));
197  keyPairGroup[index].dKey = (keyPairGroup[index - 1].dKey << (4 + shiftLen)) >> 4;
198  keyPairGroup[index].dKey |= (keyPairGroup[index - 1].dKey >> (28 - shiftLen));
199  }
200  // Assemble K_1 ~ K_16 from cKeys and dKeys.
201  uint8_t bitIndex;
202  for (index = 1; index < 17; index++) {
203  subKeys[index].key = 0x0;
204  for (bitIndex = 0; bitIndex < 48; bitIndex++) { // generate key[index]
205  pos = __permuted_key_table_PC2__[bitIndex]; // pos = which bit to get
206  if (pos < 28) { // get bit from dKey
207  bit = __getBitAt__(keyPairGroup[index].dKey, pos);
208  } else { // get bit from cKey
209  bit = __getBitAt__(keyPairGroup[index].cKey, pos - 28);
210  }
211  if (bitIndex < 16) { // set the bit in high32 part
212  __high32__(subKeys[index]) <<= 1;
213  __high32__(subKeys[index]) |= bit;
214  } else { // set the bit in low32 part
215  __low32__(subKeys[index]) <<= 1;
216  __low32__(subKeys[index]) |= bit;
217  }
218  }
219  }
220  return subKeys;
221 }

223 /**
224  * @brief Calculate the R part with the last item and the current key.
225  * @param lastItem The message data combined by the last L part and R part.
226  * @param key The current DES key, 48-bit.
227  * @return uint32_t The R part.
228  */
229 uint32_t __calculate_R_part__(MessageData lastItem, DES_Key key)
230 {
231  register uint32_t r_part = 0x0;
232  MessageData temp;
233  temp.data = 0x0;
234 
235  register uint32_t index, bit, pos, i, j;
236  // expand R part of lastItem from 32bit to 48bit
237  for (index = 0; index < 16; index++) { // high 16 bits
238  pos = __E_BIT_SELECTION_TABLE__[index];
239  __high32__(temp) <<= 1;
240  __high32__(temp) |= __getBitAt__(__R_part__(lastItem), pos);
241  }
242  for (index = 16; index < 48; index++) { // low 32 bits
243  pos = __E_BIT_SELECTION_TABLE__[index];
244  __low32__(temp) <<= 1;
245  __low32__(temp) |= __getBitAt__(__R_part__(lastItem), pos);
246  }
247  __low32__(temp) ^= __low32__(key); // XOR with the key, low32
248  __high32__(temp) ^= __high32__(key); // XOR with the key, high32
249  __high32__(temp) &= 0xFFFF; // make sure high 16bit of high32 part is 0x0.
250  // index = 0
251  bit = (__high32__(temp) & (0x3F << 10)) >> 10;
252  i = (((bit & 0x20) >> 4) | (bit & 0x1));
253  j = (bit & 0x1e) >> 1;
254  r_part <<= 4;
255  r_part |= (__S_TABLE__[0][i][j] & 0xF);
256  // index = 1
257  bit = (__high32__(temp) & (0x3F << 4)) >> 4;
258  i = (((bit & 0x20) >> 4) | (bit & 0x1));
259  j = (bit & 0x1e) >> 1;
260  r_part <<= 4;
261  r_part |= (__S_TABLE__[1][i][j] & 0xF);
262  // index = 2
263  bit = ((__high32__(temp) & 0xF) << 2) | (__low32__(temp) >> 30);
264  i = (((bit & 0x20) >> 4) | (bit & 0x1));
265  j = (bit & 0x1e) >> 1;
266  r_part <<= 4;
267  r_part |= (__S_TABLE__[2][i][j] & 0xF);
268  // index >= 3
269  for (index = 3, pos = 24; index < 8; index++, pos -= 6) { // compress R part from 48bit to 32bit by S_TABLE
270  bit = (__low32__(temp) & (0x3F << pos)) >> pos;
271  i = (((bit & 0x20) >> 4) | (bit & 0x1));
272  j = (bit & 0x1e) >> 1;
273  r_part <<= 4;
274  r_part |= (__S_TABLE__[index][i][j] & 0xF);
275  }
276  // P
277  __R_part__(temp) = r_part;
278  r_part = 0x0; // clear part
279  for (index = 0; index < 32; index++) {
280  pos = __PERMUTATION_P_TABLE__[index];
281  r_part <<= 1;
282  r_part |= __getBitAt__(__R_part__(temp), pos);
283  }
284  r_part ^= __L_part__(lastItem); // XOR L(n-1)
285  return r_part;
286 }
287 /**
288 * @brief Process data with specific subkeys and specific mode.
289 * @param originalData data to process.
290 * @param subKeys 17 keys: 1 original key and 16 subkeys.
291 * @param mode DES_ENCRYPT_MODE or DES_DECRYPT_MODE
292 * @return MessageData Processed data.
293 */
294 MessageData DES_process(MessageData originalData, DES_Key* subKeys, uint8_t mode)
295 {
296  register uint32_t index, pos;
297  MessageData output, temp;
298  // Initialize output and temp.
299  output.data = 0x0;
300  temp.data = 0x0;
301 
302  // Generate initial permuted data
303  for (index = 0; index < 32; index++) { // high 32 bits
304  pos = __permuted_message_table_IP__[index];
305  __L_part__(temp) <<= 1;
306  if (pos < 32) { // get bit from low32
307  __L_part__(temp) |= __getBitAt__(__low32__(originalData), pos);
308  } else { // get bit from high32
309  __L_part__(temp) |= __getBitAt__(__high32__(originalData), pos - 32);
310  }
311  }
312  for (index = 32; index < 64; index++) { // low 32 bits
313  pos = __permuted_message_table_IP__[index];
314  __R_part__(temp) <<= 1;
315  if (pos < 32) { // get bit from low32
316  __R_part__(temp) |= __getBitAt__(__low32__(originalData), pos);
317  } else { // get bit from high32
318  __R_part__(temp) |= __getBitAt__(__high32__(originalData), pos - 32);
319  }
320  }
321 
322  // Iteration
323  // n = [1, 16]
324  // mode = ENCRYPT_MODE = 0x0: L(n) = R(n-1), R(n) = L(n-1) + f(R(n-1), subKey(n))
325  // mode = DECRYPT_MODE = 0x1: L(n) = R(n-1), R(n) = L(n-1) + f(R(n-1), subKey(17-n))
326  // MessageData.apart[0] = R part, MessageData.apart[1] = L part
327  if (mode == DES_DECRYPT_MODE) { // Decrypt mode
328  for (index = 1; index < 17; index++) { // Iteration of L part and R part
329  __L_part__(output) = __R_part__(temp);
330  __R_part__(output) = __calculate_R_part__(temp, subKeys[17 - index]);
331  temp = output;
332  }
333  } else { // Encrypt mode
334  for (index = 1; index < 17; index++) { // Iteration of L part and R part
335  __L_part__(output) = __R_part__(temp);
336  __R_part__(output) = __calculate_R_part__(temp, subKeys[index]);
337  temp = output;
338  }
339  }
340 
341  // exchange L part and R part.
342  __L_part__(temp) = __R_part__(output);
343  __R_part__(temp) = __L_part__(output);
344  output.data = 0x0; // clear output
345  // final permutation by P-1 table
346  for (index = 0; index < 32; index ++) { // high 32 bits
347  pos = __PERMUTATION_P_1_TABLE__[index];
348  __high32__(output) <<= 1;
349  if (pos < 32) { // get bit from low32
350  __high32__(output) |= __getBitAt__(__low32__(temp), pos);
351  } else { // get bit from high32
352  __high32__(output) |= __getBitAt__(__high32__(temp), pos - 32);
353  }
354  }
355  for (index = 32; index < 64; index ++) { // low 32 bits
356  pos = __PERMUTATION_P_1_TABLE__[index];
357  __low32__(output) <<= 1;
358  if (pos < 32) { // get bit from low32
359  __low32__(output) |= __getBitAt__(__low32__(temp), pos);
360  } else { // get bit from high32
361  __low32__(output) |= __getBitAt__(__high32__(temp), pos - 32);
362  }
363  }
364  return output;
365 }
