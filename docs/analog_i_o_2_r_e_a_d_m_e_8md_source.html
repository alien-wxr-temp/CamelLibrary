96 ### Example
97 
98 ```C
99 
100 #include "analogIO.h"
101 
102 // This example setups SD simply.
103 void Example_SD_SimpleSst()
104 {
105  RT_ADC_SD_Setup(SD_CLK_3M, SD_20BIT, SD_TRIG_BY_WT2READ);
106  // Turn on SD, clear SD, start accumulation are all done in RT_ADC_SD_Read
107  uint32_t result = RT_ADC_SD_Read();
108  // ...
109 }
110 
111 // This example reveals more details about SD setup.
112 void Example_SD_Basic()
113 {
114  RT_ADC_SD_SetSampleRate(SD_CLK_3M);
115  RT_ADC_SD_SetAdWidth(SD_20BIT);
116  RT_ADC_SD_SetTrigger(SD_TRIG_BY_TC0PWM);
117  RT_ADC_Clear();
118  RT_ADC_SD_On();
119  for (register uint32_t i=0; i<200;i++)
120  __asm__("nop");
121  RT_ADC_SD_Start();
122  while(!RT_ADC_SD_DataReady());
123  uint32_t result = MemoryRead32(AD_READ_REG) & 0xfffff;
124  // ...
125 }
